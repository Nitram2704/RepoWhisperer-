```python
@app.command()
def run(
    path: str = typer.Argument(..., help="Path to source directory"),
    output_dir: str = typer.Option(None, "--output-dir", "-o", help="Output directory (default: <path>/docs)"),
):
    """Parse, index, and generate documentation for a project."""
    from docgen.config import load_config
    from docgen.ingest import run_ingest
    from docgen.parser import parse_directory
    from docgen.runner import group_chunks_by_module, generate_all_docs
    from docgen.writer import write_docs
    from docgen.store import VectorRepository
    from docgen.llm import get_provider
    from rich.progress import Progress, SpinnerColumn, TextColumn, BarColumn, TaskProgressColumn
    from rich.table import Table
    import asyncio
```

**Full flow:**

1. **Validate config:**
   ```python
   config = load_config()  # fails fast if DOCGEN_API_KEY missing
   ```

2. **Resolve paths:**
   ```python
   from pathlib import Path
   project = Path(path).resolve()
   if not project.is_dir():
       typer.echo(f"Error: {project} is not a directory.", err=True)
       raise typer.Exit(code=1)
   out = Path(output_dir) if output_dir else project / "docs"
   ```

3. **Phase 1 — Index:**
   ```python
   ingest_result = run_ingest(str(project))
   # run_ingest already shows spinners for parse/embed/store
   ```

4. **Phase 2 — Generate:**
   ```python
   provider_name = os.getenv("DOCGEN_PROVIDER", "gemini")
   provider = get_provider(provider_name)
   chunks = parse_directory(project)  # re-parse for grouping
   module_groups = group_chunks_by_module(chunks, str(project))
   repo = VectorRepository(chroma_dir=str(project / ".chroma"))

   total = 1 + len(module_groups)  # 1 README + N modules
   with Progress(
       SpinnerColumn(),
       TextColumn("[progress.description]{task.description}"),
       BarColumn(),
       TaskProgressColumn(),
   ) as progress:
       task_id = progress.add_task("Generating docs...", total=total)
       readme, module_docs = asyncio.run(
           generate_all_docs(
               module_groups, repo, provider, provider_name,
               on_progress=lambda: progress.advance(task_id),
           )
       )
   ```

5. **Phase 3 — Write:**
   ```python
   written = write_docs(readme, module_docs, out, project)
   ```

6. **Phase 4 — Summary:**
   ```python
   table = Table(title="DocGen Summary")
   table.add_column("Metric", style="cyan")
   table.add_column("Value", style="green")
   table.add_row("Files parsed", str(ingest_result["parsed"]))
   table.add_row("Chunks indexed", str(ingest_result.get("upserted", 0)))
   table.add_row("Chunks skipped", str(ingest_result.get("skipped", 0)))
   table.add_row("Docs generated", str(len(written)))
   table.add_row("Output directory", str(out))
   console.print(table)
   ```

**Important constraints:**
- `load_config()` MUST be called before any LLM usage — it validates the API key
- `parse_directory` is called twice: once inside `run_ingest` (for embedding) and once in `run` (for grouping). This is acceptable because parsing is fast (<1s for 50 files). Alternative: modify `run_ingest` to return chunks, but that changes its contract from Phase 3.
- Default output is `<project>/docs` (not CWD/docs) — per 05-RESEARCH.md Pitfall 3

```python
@app.command()
def generate(
    output_dir: str = typer.Option(None, "--output-dir", "-o", help="Output directory (default: ./docs)"),
    chroma_dir: str = typer.Option(".chroma", "--chroma-dir", help="Path to existing .chroma index"),
):
    """Generate docs from an existing vector index (no re-parsing)."""
    from docgen.config import load_config
    from docgen.store import VectorRepository
    from docgen.runner import file_path_to_module_name, generate_all_docs
    from docgen.writer import write_docs
    from docgen.llm import get_provider
    from docgen.models import CodeChunk
    from rich.progress import Progress, SpinnerColumn, TextColumn, BarColumn, TaskProgressColumn
    from rich.table import Table
    import asyncio
    from pathlib import Path

    config = load_config()

    # Verify index exists
    chroma_path = Path(chroma_dir).resolve()
    if not chroma_path.is_dir():
        typer.echo(
            f"Error: No index found at {chroma_path}. "
            "Run `docgen index <path>` first.",
            err=True,
        )
        raise typer.Exit(code=1)

    repo = VectorRepository(chroma_dir=str(chroma_path))
    if repo.count() == 0:
        typer.echo("Error: Index is empty. Run `docgen index <path>` first.", err=True)
        raise typer.Exit(code=1)

    # Recover module list from stored metadata
    file_paths = repo.list_files()
    # Derive source root from common ancestor of stored paths
    source_root = str(Path(os.path.commonpath(file_paths)).parent) if file_paths else "."

    # Build module groups from file_paths (without re-parsing)
    module_groups: dict[str, list] = {}
    for fp in file_paths:
        module_name = file_path_to_module_name(fp, source_root)
        module_groups[module_name] = []  # empty list — generate_docs queries the repo directly

    # Generate
    out = Path(output_dir) if output_dir else Path("docs")
    provider_name = os.getenv("DOCGEN_PROVIDER", "gemini")
    provider = get_provider(provider_name)

    total = 1 + len(module_groups)
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        BarColumn(),
        TaskProgressColumn(),
    ) as progress:
        task_id = progress.add_task("Generating docs...", total=total)
        readme, module_docs = asyncio.run(
            generate_all_docs(
                module_groups, repo, provider, provider_name,
                on_progress=lambda: progress.advance(task_id),
            )
        )

    # Write — use source_root as source_dir for validation
    written = write_docs(readme, module_docs, out, Path(source_root))

    # Summary
    table = Table(title="DocGen Generate Summary")
    table.add_column("Metric", style="cyan")
    table.add_column("Value", style="green")
    table.add_row("Modules in index", str(len(file_paths)))
    table.add_row("Docs generated", str(len(written)))
    table.add_row("Output directory", str(out))
    console.print(table)
```

**Key difference from `run`:**
- No `parse_directory` or `run_ingest` call — uses existing index
- Module list comes from `repo.list_files()` (added in Plan 01 Task 3)
- Source root is derived from `os.path.commonpath()` of stored file paths
- `module_groups` values are empty lists (generate_docs queries the repo directly by text, not by pre-loaded chunks)
