4. Create `src/docgen/llm/retry.py` with TWO retry decorators:

```python
from tenacity import retry, retry_if_exception_type, stop_after_attempt, wait_exponential_jitter

# For Groq, OpenRouter, DeepSeek (all use openai SDK)
from openai import RateLimitError as OpenAIRateLimitError, APITimeoutError

openai_retry = retry(
    retry=retry_if_exception_type((OpenAIRateLimitError, APITimeoutError)),
    wait=wait_exponential_jitter(initial=2, max=60),
    stop=stop_after_attempt(5),
    reraise=True,
)

# For Gemini (uses google-generativeai SDK)
from google.api_core.exceptions import ResourceExhausted, DeadlineExceeded

gemini_retry = retry(
    retry=retry_if_exception_type((ResourceExhausted, DeadlineExceeded)),
    wait=wait_exponential_jitter(initial=2, max=60),
    stop=stop_after_attempt(5),
    reraise=True,
)
```

Both decorators use `initial=2` (2-second first wait), `max=60` (cap at 60 seconds), 5 attempts, jitter to avoid thundering herd. `reraise=True` so the final error propagates cleanly.

1. Create `src/docgen/llm/groq.py`:
```python
import os
from openai import OpenAI
from docgen.llm.base import BaseLLMProvider
from docgen.llm.retry import openai_retry

class GroqProvider(BaseLLMProvider):
    BASE_URL = "https://api.groq.com/openai/v1"
    DEFAULT_MODEL = "llama-3.3-70b-versatile"
    ENV_KEY = "GROQ_API_KEY"

    def __init__(self, model: str | None = None):
        api_key = os.environ.get(self.ENV_KEY)
        if not api_key:
            raise SystemExit(f"Missing {self.ENV_KEY} environment variable. Get one at https://console.groq.com/")
        self.model = model or os.getenv("DOCGEN_MODEL", self.DEFAULT_MODEL)
        self._client = OpenAI(api_key=api_key, base_url=self.BASE_URL)

    @openai_retry
    def generate(self, system: str, user: str) -> str:
        response = self._client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": system},
                {"role": "user", "content": user},
            ],
        )
        return response.choices[0].message.content
```

4. Create `src/docgen/llm/gemini.py` — uses google-generativeai SDK, NOT openai:
```python
import os
import google.generativeai as genai
from docgen.llm.base import BaseLLMProvider
from docgen.llm.retry import gemini_retry

class GeminiProvider(BaseLLMProvider):
    DEFAULT_MODEL = "gemini-2.5-flash"
    ENV_KEY = "GEMINI_API_KEY"

    def __init__(self, model: str | None = None):
        api_key = os.environ.get(self.ENV_KEY) or os.environ.get("GOOGLE_API_KEY")
        if not api_key:
            raise SystemExit(f"Missing {self.ENV_KEY} environment variable. Get one at https://aistudio.google.com/")
        genai.configure(api_key=api_key)
        self.model = model or os.getenv("DOCGEN_MODEL", self.DEFAULT_MODEL)
        self._model = genai.GenerativeModel(
            model_name=self.model,
            system_instruction=None,  # Set per-call
        )

    @gemini_retry
    def generate(self, system: str, user: str) -> str:
        # Gemini GenerativeModel supports system_instruction at construction
        # but we set it per-call by creating a new model instance or using
        # generate_content with a system message. Simplest approach:
        model = genai.GenerativeModel(
            model_name=self.model,
            system_instruction=system,
        )
        response = model.generate_content(user)
        return response.text
```

IMPORTANT: The `gemini_retry` decorator catches `google.api_core.exceptions.ResourceExhausted` for 429 rate limits. If during testing the actual exception class differs, update retry.py accordingly. The research flagged this as MEDIUM confidence — see Open Question 3 in 04-RESEARCH.md.

```python
"""LLM provider package. Use get_provider() to get the active provider."""
import os
from docgen.llm.base import BaseLLMProvider

def get_provider(provider_name: str | None = None, model: str | None = None) -> BaseLLMProvider:
    """Return an LLM provider instance based on config.

    Provider is selected by (in priority order):
    1. provider_name argument (if given)
    2. DOCGEN_PROVIDER env var
    3. Default: "gemini"

    Model is selected by (in priority order):
    1. model argument (if given)
    2. DOCGEN_MODEL env var
    3. Provider's DEFAULT_MODEL
    """
    from docgen.llm.gemini import GeminiProvider
    from docgen.llm.groq import GroqProvider
    from docgen.llm.openrouter import OpenRouterProvider
    from docgen.llm.deepseek import DeepSeekProvider

    providers = {
        "gemini": GeminiProvider,
        "groq": GroqProvider,
        "openrouter": OpenRouterProvider,
        "deepseek": DeepSeekProvider,
    }

    name = (provider_name or os.getenv("DOCGEN_PROVIDER", "gemini")).lower().strip()
    if name not in providers:
        raise SystemExit(
            f"Unknown DOCGEN_PROVIDER={name!r}. Valid providers: {', '.join(sorted(providers))}"
        )
    return providers[name](model=model)

__all__ = ["get_provider", "BaseLLMProvider"]
```

Note: Imports are INSIDE get_provider() (lazy) so that importing the package does not require all SDK env vars to be set. Only the selected provider's env var is needed.
